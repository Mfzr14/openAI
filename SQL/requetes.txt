Creation vue avec jointure

CREATE VIEW `vue avecd jointure` AS
SELECT c.id, c.name, un.name as universeName, u.id as userID
FROM `characters` `c`
JOIN `users` `u` ON c.user_id  = u.id 
JOIN `universe` un ON un.user_id = u.id
